1function[yn,W,en]=RLS(xn,dn,M,itr)
 2% RLS(Least Mean Squre)算法
 3% 输入参数:
 4%     xn   输入的信号序列      (行向量)
 5%     dn   所期望的响应序列    (行向量)
 6%     M    滤波器的阶数        (标量)
 7%     itr  迭代次数            (标量)     默认为xn的长度,M<itr<length(xn)
 8% 输出参数:
 9%     W    滤波器的权值向量     (行向量)
10%          大小为M
11%     en   误差序列(itr x 1)    (行向量)  
12%     yn   实际输出序列         (行向量)
13
14% 参数个数必须为4个或5个
15if nargin == 3                 % 4个时递归迭代的次数为xn的长度 
16    itr = length(xn);
17elseif nargin == 4             % 5个时满足M<itr<length(xn)
18    if itr>length(xn) || itr<M
19        itr = length(xn);
20    end
21else
22    error('please check the input number num!!');
23end
24
25
26% 初始化参数
27en = zeros(itr,1);             % 误差序列,en(n)表示第k次迭代时预期输出与实际输入的误差
28W  = zeros(1,M);             % 每一行代表一个加权参量,每一列代表-次迭代,初始为0
29yn = zeros(1,itr);
30lamda = 0.98;
31p=0.1*eye(M,M);
32% 迭代计算
33for n = M:itr                  % 第k次迭代
34    x = xn(n:-1:n-M+1)';        % 滤波器M个抽头的输入
35%     y = W(:,n-1)' * x';        % 滤波器的输出
36    k = ((1/lamda)*p*x)/(1+(1/lamda)*x'*p*x);
37    en(n) = dn(n)-W*x;
38    W=W+(k*conj(en(n))')';
39    p=(1/lamda)*p-(1/lamda)*k*x'*p;
40    yn(n)=W*x;
41
42end
