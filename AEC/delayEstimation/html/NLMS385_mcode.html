1function[yn,W,en]=NLMS(xn,dn,M,mu,itr)
 2% NLMS(Normalized Least Mean Squre)算法
 3% 输入参数:
 4%     xn   输入的信号序列      (行向量)
 5%     dn   所期望的响应序列    (行向量)
 6%     M    滤波器的阶数        (标量)
 7%     mu   收敛因子(步长)      (标量)     要求大于0,小于xn的相关矩阵最大特征值的倒数    
 8%     itr  迭代次数            (标量)     默认为xn的长度,M<itr<length(xn)
 9% 输出参数:
10%     W    滤波器的权值向量     (行向量)
11%          大小为M
12%     en   误差序列(itr x 1)    (行向量)  
13%     yn   实际输出序列         (行向量)
14
15% 参数个数必须为4个或5个
16if nargin == 4                 % 4个时递归迭代的次数为xn的长度 
17    itr = length(xn);
18elseif nargin == 5             % 5个时满足M<itr<length(xn)
19    if itr>length(xn) || itr<M
20        itr = length(xn);
21    end
22else
23    error('please check the input number num!!');
24end
25
26
27% 初始化参数
28en = zeros(itr,1);             % 误差序列,en(k)表示第k次迭代时预期输出与实际输入的误差
29W  = zeros(1,M);             % 每一行代表一个加权参量,每一列代表-次迭代,初始为0
30yn = zeros(1,itr);
31
32% 迭代计算
33for k = M:itr                  % 第k次迭代
34    x = xn(k:-1:k-M+1);        % 滤波器M个抽头的输入
35    yn(k) = W * x';        % 滤波器的输出
36    en(k) = dn(k) - yn(k) ;        % 第k次迭代的误差
37    
38    % 滤波器权值计算的迭代式
39    W = W + mu*en(k)*x/(1+x*x');
40end
